  Creates invite link for channel

  @param {Array.<Array.<*>>} options - Options for invites
*/
public function createInvite(temp.options) {
  temp.req = this._super.makeRequest("post", format(this._super.endpoints.channel.invites, this.id), temp.options, false);
  if (!this._super.checkForError(temp.req)) {
    temp.req.join("discord_channel_guild_invite");
    return temp.req;
  }  
}
