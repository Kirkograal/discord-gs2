function onCreated() {
  this.clearVars();

  // "T" is not needed but it looks cool
  createClass("TDiscord", "TStaticVar", {"discord", "discord_embed", "discord_http_endpoints", "discord_http", "func_http"});
  createClass("TCollection", "TStaticVar", {"discord_collection"});
  createClass("TWebhookClient", "TStaticVar", {"discord_webhook_client", "discord_embed", "discord_http_endpoints", "discord_http", "func_http", "func_promises"});

  RC.showErrors();
}

function createClass(className, type, extendedClasses) {
  if (isObject(@ classNasame))
    (@ classNasame).destroy();

  this.(@ className) = new (@ type)(@ className);
  for (temp.c : extendedClasses)
    this.(@ className).join(temp.c);
}
